⚡ ما هو JavaScript؟

JavaScript

لغة برمجة تُستخدم لإضافة التفاعل والحركة والذكاء لصفحات الويب

إذا كان HTML هو الهيكل العظمي و CSS هو الجلد والملابس، فإن JavaScript هو الدماغ والعضلات! JavaScript يجعل موقعك حياً ومتفاعلاً.

تشبيه مفيد: تخيل موقع الويب كسيارة - HTML هو الهيكل، CSS هو الطلاء والتصميم، و JavaScript هو المحرك الذي يجعلها تتحرك!

ماذا يمكن لـ JavaScript أن يفعل؟

  • تغيير المحتوى: إضافة أو حذف أو تعديل النصوص والصور
  • التفاعل مع المستخدم: الاستجابة للنقرات والكتابة
  • التحقق من البيانات: فحص النماذج قبل الإرسال
  • الحركة والتأثيرات: إنشاء رسوم متحركة
  • التواصل مع الخادم: جلب البيانات دون إعادة تحميل الصفحة

🧮 العمليات (Operations)

العمليات الحسابية:

العمليات الحسابية 
let a = 10;
let b = 3;

console.log(a + b);  // الجمع: 13
console.log(a - b);  // الطرح: 7
console.log(a * b);  // الضرب: 30
console.log(a / b);  // القسمة: 3.333...
console.log(a % b);  // باقي القسمة: 1
console.log(a ** b); // الأس: 1000

عمليات المقارنة:

المقارنات 
let x = 5;
let y = "5";

console.log(x == y);   // true (مقارنة القيمة فقط)
console.log(x === y);  // false (مقارنة القيمة والنوع)
console.log(x != y);   // false
console.log(x !== y);  // true
console.log(x > 3);    // true
console.log(x <= 5);   // true

العمليات المنطقية:

العمليات المنطقية 
let isAdult = true;
let hasLicense = false;

console.log(isAdult && hasLicense); // AND: false
console.log(isAdult || hasLicense); // OR: true
console.log(!isAdult);              // NOT: false

🔧 الدوال (Functions)

الدوال هي قطع من الكود يمكن إعادة استخدامها:

إنشاء دالة:

دالة بسيطة 
// دالة بدون معاملات
function sayHello() {
    alert("مرحباً!");
}

// دالة مع معاملات
function greet(name) {
    return "مرحباً " + name + "!";
}

// دالة مع معاملات متعددة
function add(a, b) {
    return a + b;
}

// استخدام الدوال
sayHello();
let message = greet("أحمد");
let sum = add(5, 3);

الدوال السهمية (Arrow Functions):

Arrow Functions 
// الطريقة التقليدية
function multiply(a, b) {
    return a * b;
}

// الدالة السهمية
const multiply = (a, b) => {
    return a * b;
}

// دالة سهمية مختصرة
const multiply = (a, b) => a * b;

// دالة سهمية بمعامل واحد
const square = x => x * x;

🤔 الشروط (Conditions)

if...else:

الشروط 
let age = 18;

if (age >= 18) {
    console.log("بالغ");
} else if (age >= 13) {
    console.log("مراهق");
} else {
    console.log("طفل");
}

// الشرط المختصر (Ternary)
let status = age >= 18 ? "بالغ" : "قاصر";

// Switch
let day = "الاثنين";
switch (day) {
    case "السبت":
    case "الأحد":
        console.log("عطلة");
        break;
    case "الاثنين":
        console.log("بداية الأسبوع");
        break;
    default:
        console.log("يوم عادي");
}

🔄 الحلقات (Loops)

أنواع الحلقات:

الحلقات 
// for loop
for (let i = 1; i <= 5; i++) {
    console.log("الرقم: " + i);
}

// while loop
let count = 1;
while (count <= 5) {
    console.log("العد: " + count);
    count++;
}

// for...of (للمصفوفات)
let fruits = ["تفاح", "موز", "برتقال"];
for (let fruit of fruits) {
    console.log(fruit);
}

// for...in (للكائنات)
let person = {name: "أحمد", age: 25, city: "الرياض"};
for (let key in person) {
    console.log(key + ": " + person[key]);
}

🌐 التعامل مع DOM

DOM (Document Object Model) يتيح لك تغيير محتوى الصفحة:

اختيار العناصر:

اختيار العناصر 
// اختيار بالـ ID
let element = document.getElementById("myElement");

// اختيار بالفئة
let elements = document.getElementsByClassName("myClass");

// اختيار بالعنصر
let paragraphs = document.getElementsByTagName("p");

// اختيار بـ CSS Selector
let element = document.querySelector("#myElement");
let elements = document.querySelectorAll(".myClass");

تغيير المحتوى:

تغيير المحتوى 
// تغيير النص
element.textContent = "نص جديد";
element.innerHTML = "<strong>نص غامق</strong>";

// تغيير الخصائص
element.style.color = "red";
element.style.fontSize = "20px";

// إضافة/إزالة فئات
element.classList.add("newClass");
element.classList.remove("oldClass");
element.classList.toggle("activeClass");

🎯 الأحداث (Events)

أنواع الأحداث:

الأحداث 
// النقر
button.addEventListener("click", function() {
    alert("تم النقر!");
});

// تحميل الصفحة
window.addEventListener("load", function() {
    console.log("تم تحميل الصفحة");
});

// الكتابة في حقل النص
input.addEventListener("input", function() {
    console.log("تم تغيير النص");
});

// تمرير الماوس
element.addEventListener("mouseover", function() {
    this.style.backgroundColor = "yellow";
});

element.addEventListener("mouseout", function() {
    this.style.backgroundColor = "";
});

🧠 اختبر فهمك

1. ما هو الفرق بين let و const؟

  • لا يوجد فرق
  • let قابل للتغيير، const ثابت
  • const أسرع من let
  • let للنصوص، const للأرقام

2. كيف نختار عنصر له id="myButton"؟

  • document.getElementById("myButton")
  • document.getClass("myButton")
  • document.select("myButton")
  • getElementById("myButton")

3. ما هو DOM؟

  • نوع من أنواع البيانات
  • Document Object Model
  • لغة برمجة جديدة
  • نوع من أنواع الدوال

4. كيف نضيف مستمع حدث للنقر؟

  • element.click(function)
  • element.addEventListener("click", function)
  • element.onClick(function)
  • element.addClick(function)
