🐍 ما هو Python؟

Python

لغة برمجة عالية المستوى، سهلة التعلم، قوية ومتعددة الاستخدامات

Python هي واحدة من أشهر لغات البرمجة في العالم! تُستخدم في كل شيء تقريباً: مواقع الويب، الذكاء الاصطناعي، تحليل البيانات، الألعاب، وحتى في ناسا!

لماذا Python؟ لأنها بسيطة مثل الإنجليزية، قوية مثل السحر، ومطلوبة في سوق العمل!

لماذا نتعلم Python؟

  • سهولة التعلم: بناء جملة بسيط وواضح
  • قوة الاستخدام: يمكن بناء أي شيء تقريباً
  • مجتمع كبير: ملايين المطورين حول العالم
  • مكتبات ضخمة: آلاف المكتبات الجاهزة
  • فرص عمل: من أكثر اللغات طلباً في السوق

guess_game.py 
import random

def guess_number_game():
    print("🎯 مرحباً بك في لعبة تخمين الرقم!")
    print("سأفكر في رقم بين 1 و 100")
    print("حاول أن تخمنه في أقل عدد من المحاولات!")
    
    secret_number = random.randint(1, 100)
    attempts = 0
    max_attempts = 7
    
    while attempts < max_attempts:
        try:
            guess = int(input(f"\nالمحاولة {attempts + 1}/{max_attempts} - أدخل تخمينك: "))
            attempts += 1
            
            if guess < 1 or guess > 100:
                print("يرجى إدخال رقم بين 1 و 100")
                continue
            
            if guess == secret_number:
                print(f"🎉 مبروك! لقد خمنت الرقم {secret_number} في {attempts} محاولة!")
                return attempts
            elif guess < secret_number:
                print("📈 الرقم أكبر من تخمينك")
            else:
                print("📉 الرقم أصغر من تخمينك")
                
        except ValueError:
            print("يرجى إدخال رقم صحيح")
    
    print(f"😞 انتهت المحاولات! الرقم كان {secret_number}")
    return max_attempts

# تشغيل اللعبة
if __name__ == "__main__":
    while True:
        score = guess_number_game()

        if play_again.lower() != 'y':
            print("شكراً لك على اللعب! 👋")
            break
